function onChange(control, oldValue, newValue, isLoading, isTemplate) {
   if (isLoading || newValue === '') {
      return;
   }

  var ga = new GlideAjax('ShippingCaseUtils');
  ga.addParam('sysparm_name', 'validateDOB');
  ga.addParam('sysparm_date', g_form.getValue('u_dob'));
  ga.getXMLAnswer(callback);

  function callback (res) {
	if (res == 'no') {
		g_form.clearValue('u_dob');
		g_form.showFieldMsg('u_dob', 'DOB cannot be a future date', 'error');
	}
  }
}
