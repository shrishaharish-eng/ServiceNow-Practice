function validateForm() {
	var resolutionCode = gel('rcode').value;
	var resolutionNotes = gel('rnotes').value;

	if (resolutionCode.trim() == '') {
		alert('Resolution Code is mandatory');
		return false;
	}

	if (resolutionNotes.trim() == '') {
		alert('Resolution Notes is mandatory');
		return false;
	}

	var ga = new GlideAjax('ShippingCaseUtils');
		ga.addParam('sysparm_name', 'resolveShippingCase');
		ga.addParam('sysparm_scid', g_form.getUniqueValue());
		ga.addParam('sysparm_rcode', resolutionCode);
		ga.addParam('sysparm_rnotes', resolutionNotes);
		ga.getXMLAnswer(callback);

		function callback(res) {
			if (res == 'true') {
				alert('Case has been resolved successfully');
				parent.window.location.reload();
				GlideDialogWindow.get().destroy();
			}
		}

}

function onCancel() {
	GlideDialogWindow.get().destroy();
}
