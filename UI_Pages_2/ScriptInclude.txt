resolveShippingCase: function() {
		var gr = new GlideRecord("u_shipping_case");
		gr.addQuery("sys_id", this.getParameter('sysparm_scid'));
		gr.query();
		var isUpdated = false;
		if (gr.next()) {
			gr.u_resolution_code = this.getParameter('sysparm_rcode');
			gr.u_resolution_notes = this.getParameter('sysparm_rnotes');
			gr.u_state = 'Resolved';
			gr.update();
			isUpdated = true;
		}

		return isUpdated;
	},
